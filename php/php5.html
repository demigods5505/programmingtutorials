    <?php


    class Player{
        //class variables
        public $speed = 5 ; 
        public $running = false ;  
    
        //methods
        //functions are public by defauult
        function set_name($naming){
           $this->name=$naming;  //name is instance variable
        }
    
        function get_name(){
            return $this->name; //printing instance variable
        }
    
        function run(){
           $this->running=true; //this is also a instance variable
        }
    
        function stop(){
           $this->running=false; 
        }
    }
    
    echo "lets understand opps using gta vice city<br>";
    $player1= new player(); //new object/instance created using class name Player
    $player1->set_name("rohan");
    echo $player1->get_name();
    $player1->speed=88;//setting instance variable if not set then class variable is printed
    echo $player1->speed;
    echo "<br>";
    $player2= new player();
    $player2->set_name("harry");
    echo $player2->get_name();
    echo $player2->speed; //here instance variable is not present so class variable is accessed
    echo "<br>";
    $player3= new player();
    $player3->set_name("sagar");
    echo $player3->get_name();
    
    ?>
