types of selectors 

    <!DOCTYPE html>
    <html lang="en">
    <head>
        <meta charset="UTF-8">
        <meta name="viewport" content="width=device-width, initial-scale=1.0">
        <meta http-equiv="X-UA-Compatible" content="ie=edge">
        <title>types of selectors</title>
        <!-- for id selector use # and for class selector use . -->
        <style>
            /* element selector */
            p{
                border: 2px solid black;
            }
            /* id selector */
            #myname{
                color: blue;
                background-color: bisque;
            }
            /* class selector, here we use 2 classes therefore the last class is preceding in text color */
            .westgreen{
                color: rgb(11, 206, 164);
                background-color: yellowgreen;
            }
            .indo{
                color:red;
            }
            /* grouping selector */
            span,footer{
                color: rgb(233, 38, 168);
                background-color: rgb(13, 108, 218)
            }
        </style>
    </head>
    <body>
       <h2>types of selector</h2> 
       <p id="myname">this is first paragraph</p>
       <p id="pyramid" class="indo westgreen">this is second paragraph</p>
       <p>this is third paragraph</p>
       <span>this is span</span>
       <footer>this is footer</footer>
       <!-- if u want to change styles of particular element then use id selector and id id =s given to that element -->
       <!-- for group of elements then use class and class is given to all elements  -->
    </body>
    </html>
