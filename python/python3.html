    #coroutine

    def searcher():
        import time
        book="this is a book on harry and code with harry and good"
        time.sleep(4) #assume reading a big book or doing some ml work
        print("search is started") # it only prints one time whike starting the searcher
        while True:
            text=(yield)
            if text in book:
                print("your text is found in the book")
            else:
                print("text is not found in the book")
    
    search=searcher()
    next(search) # to start the searcher and then no need of starting it it will keep its pointer befor while when anything send to it
    search.send("harry") # sending to searcher
    search.send("book on harry") # if again run no time lag/no delay
    search.send(input("enter text from the book\n")) #no delay as pointer always remain before while
    
    search.close() # to free all resources use close
