recursion 

    # recursion means function ka andar function use karna
    # this below code is not a recursive functiion
    def factorial_iterative(n):
        """function to find factorial"""
        fact=1
        for i in range(n): #means i starts from 0 to n-1
            fact=fact*(i+1)
        return fact
    
    def factorial_recursive(n):
        """functuion to find factorial"""
        if n==1:
            return 1
        else:
            return n * factorial_recursive(n-1)
    
    
    factorial=int(input("enter the number to find factorial\n"))
    print("factorial using iterrative method",factorial_iterative(factorial))
    print("factorial using recursion method",factorial_recursive(factorial))
