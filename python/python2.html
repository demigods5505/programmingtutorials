        def __init__(self,fname,lname):
            self.fname=fname

        @property #to acces this as a instance variable
        def email(self):
            if self.fname==None or self.lname==None:
                return "email is not set,please set it using setter"
            return f"{self.fname}.{self.lname}@gmail.com"

        @email.deleter
        def email(self):
            self.fname=None

    class Shape(ABC):
        @abstractmethod #use this above the method which u want to be included as a child class method
        def printarea(self):
            return 0
    #agarr koi bhi class metaclass se inherit hoti ha to vo order de sakti child classes ko kuch methods add karna ka lia
    
    class Rectangle(Shape):
        type="Rectangle"
        sides=4
    
        def __init__(self):
            self.length=6
            self.breadth=7
    
        def printarea(self):
             return self.length * self.breadth
